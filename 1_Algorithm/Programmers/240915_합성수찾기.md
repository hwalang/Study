```cpp
#include <cmath>

using namespace std;

bool is_composite(int num) 
{
    if (num <= 1) return false;
    
    // 하나라도 있으면 약수가 3개 이상이다
    for (int i = 2; i <= sqrt(num); ++i) {
        if (num % i == 0) return true;
    }
    
    return false;
}

int solution(int n) {
    // n까지의 자연수를 하나씩 합성수인지 판단한다.
    // 합성수는 bool 함수를 통해 파악
    // 합성수라면 cnt를 하나 늘린다.
    int cnt = 0;
    for (int i = 1; i <= n; ++i) {
        if (is_composite(i)) ++cnt;
    }
    return cnt;
}
```
